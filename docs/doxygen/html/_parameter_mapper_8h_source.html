109  // realtime safe
110  void markForRemoval(MappingPairType* m)
111  {
112  deletePoolWrite = m;
113  ++deletePoolWrite;
114  if (deletePoolWrite == DeletePool.end()) deletePoolWrite = DeletePool.begin();
115  // note that overflows here aren't thrown or asserted, just don't let that happen :3
116  }

119  // not realtime safe!
120  int removeAll()
121  {
122  int count = 0;
123  while (deletePoolRead != deletePoolWrite)
124  {
125  delete deletePoolRead->load().range;
126  delete deletePoolRead;
127  ++deletePoolRead;
128  if (deletePoolRead == DeletePool.end()) deletePoolRead = DeletePool.begin();
129  ++count;
130  }
131  return count;
132  }

163  inline void Process (juce::MidiBuffer& buffer) noexcept
164  {
165  for (auto a : buffer)
166  {
167  temp_m = a.getMessage();
168 
169  if (temp_m.isController())
170  {
171  temp_cc = temp_m.getControllerNumber();
172  temp_ch = temp_m.getChannel();
173 
174  for (auto m = Mappings.begin() + (temp_cc * temp_ch); m != Mappings.begin() + (temp_cc * temp_ch) + MAX_MAPPINGS_PER_PARAMETER; ++m)
175  {
176  temp_mapping = m->load();
177  if (temp_mapping.param != nullptr)
178  {
179  temp_mapping.param->beginChangeGesture();
180  temp_mapping.param->setValueNotifyingHost( temp_mapping.range->convertTo0to1(
181  map_from_cc.convertTo0to1(
182  static_cast<float>(temp_m.getControllerValue()))));
183  temp_mapping.param->endChangeGesture();
184  }
185  }
186  }
187  }
188  }
